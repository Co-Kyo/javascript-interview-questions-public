手写 call / apply / bind — 引导式讲解

10 步从零到完整实现 · 逐步引导 · 可运行验证

这道题在解决什么问题？

你有个函数 greet()，想让它对不同的人说"你好"。但 this 指向谁，取决于你怎么调用它——obj.greet() 时 this 是 obj，但 greet() 单独调用时 this 就跑了。

这就像你有一封通用的信，需要"寄给不同的人"——信的内容不变，但落款要跟着收件人变。call、apply、bind 就是让你控制函数里 this 指向谁的三种方式。

Step 1: 最小尝试 — 让 this 指向一个对象

困境：函数单独调用时 this 不指向目标对象，但 obj.fn() 时 this 就是 obj——能不能利用这个规则？
解法：把函数挂到对象上，调用（触发隐式绑定），然后删掉临时属性。三步走：挂 → 调 → 删。
function myCall(fn, obj) {
  obj.temp = fn;
  const result = obj.temp();
  delete obj.temp;
  return result;
}

// 测试
function greet() { return this.name; }
const person = { name: 'Alice' };
console.log(myCall(greet, person));
运行后观察输出，应显示 "Alice" ✅

Step 2: Symbol — 临时键冲突问题

困境：如果 obj 上本来就有 temp 属性呢？覆盖了原有属性，delete 后也没了——数据丢失！
💭 怎么保证临时键永远不会和已有属性冲突？
解法：用 Symbol('fn') 创建唯一键。每次调用都生成全局唯一的值，不可能和任何已有属性重名。
function myCall(fn, obj) {
  const fnKey = Symbol('fn');
  obj[fnKey] = fn;
  const result = obj[fnKey]();
  delete obj[fnKey];
  return result;
}

// 测试：obj 上有 temp 属性也不冲突
const obj = { name: 'Bob', temp: 'important' };
function greet() { return this.name; }
console.log(myCall(greet, obj));
console.log(obj.temp); // 原属性不受影响
输出 "Bob" 和 "important"，原属性完好 ✅

Step 3: null 和原始值处理

困境：myCall(fn, null) 报错——null 上不能挂属性。myCall(fn, 42) 也挂不上去。
💭 null 和数字这些"不能挂属性"的值，原生 call 是怎么处理的？
解法：① == null 同时捕获 null 和 undefined，回退到 globalThis；② Object() 把原始值转为包装对象，就能挂属性了。
function myCall(fn, obj) {
  obj = obj == null ? globalThis : Object(obj);
  const fnKey = Symbol('fn');
  obj[fnKey] = fn;
  const result = obj[fnKey]();
  delete obj[fnKey];
  return result;
}

// 测试：null 和原始值都能工作
function greet() { return typeof this; }
console.log(myCall(greet, null));   // "object" (globalThis)
console.log(myCall(greet, 42));     // "object" (Number 包装)
null 和 42 都不报错 ✅

Step 4: myCall — 改成原型方法

困境：现在是独立函数 myCall(fn, obj)，但原生用法是 fn.myCall(obj)——fn 应该是 this 而不是参数。
💭 fn.myCall(obj) 中，函数内部的 this 指向谁？
解法：挂到 Function.prototype 上，用 this 代替 fn 参数。加上剩余参数 ...args 支持传参。
Function.prototype.myCall = function (thisArg, ...args) {
  thisArg = thisArg == null ? globalThis : Object(thisArg);
  const fnKey = Symbol('fn');
  thisArg[fnKey] = this;
  const result = thisArg[fnKey](...args);
  delete thisArg[fnKey];
  return result;
};

// 测试
function greet(g, p) { return g + ', ' + this.name + p; }
const person = { name: 'Alice' };
console.log(greet.myCall(person, 'Hello', '!'));
输出 "Hello, Alice!" ✅

Step 5: myApply — 参数是数组

困境：apply 和 call 几乎一样，但第二个参数是数组。而且需要校验——传了非数组要抛 TypeError。
解法：复制 myCall 逻辑，改两处：① 参数从数组展开；② 加 Array.isArray 类型校验。
Function.prototype.myApply = function (thisArg, argsArray) {
  thisArg = thisArg == null ? globalThis : Object(thisArg);
  if (argsArray != null && !Array.isArray(argsArray)) {
    throw new TypeError('second argument must be an array');
  }
  const fnKey = Symbol('fn');
  thisArg[fnKey] = this;
  const result = argsArray
    ? thisArg[fnKey](...argsArray)
    : thisArg[fnKey]();
  delete thisArg[fnKey];
  return result;
};

// 测试
function greet(g, p) { return g + ', ' + this.name + p; }
const person = { name: 'Alice' };
console.log(greet.myApply(person, ['Hi', '~']));
输出 "Hi, Alice~" ✅

Step 6: myBind — 返回延迟执行的新函数

困境：bind 不立即执行，而是返回一个新函数。新函数需要"记住"原始函数、绑定的 this 和预设参数。
💭 新函数需要记住哪三样东西？
解法：闭包记住 originalFn、thisArg、presetArgs。新函数被调用时合并预设参数和后续参数执行。
Function.prototype.myBind = function (thisArg, ...presetArgs) {
  const originalFn = this;
  const bound = function (...laterArgs) {
    const fnKey = Symbol('fn');
    thisArg[fnKey] = originalFn;
    const result = thisArg[fnKey](...presetArgs, ...laterArgs);
    delete thisArg[fnKey];
    return result;
  };
  return bound;
};

// 测试
function greet(g, p) { return g + ', ' + this.name + p; }
const person = { name: 'Alice' };
const boundGreet = greet.myBind(person, 'Hey');
console.log(boundGreet('?'));
输出 "Hey, Alice?" ✅

Step 7: new 调用时 this 不该被绑定

困境：new BoundPerson(25) 时，JS 创建新对象作为 this，但 bound 函数硬编码用 thisArg，new 就失效了——属性全挂到绑定对象上。
💭 怎么区分"普通调用"和 new 调用？
解法：this instanceof bound 为 true 说明是 new 调用，用 this（新实例）；否则用 thisArg。new 优先级高于 bind。
Function.prototype.myBind = function (thisArg, ...presetArgs) {
  const originalFn = this;
  const bound = function (...laterArgs) {
    const context = this instanceof bound ? this : thisArg;
    const fnKey = Symbol('fn');
    context[fnKey] = originalFn;
    const result = context[fnKey](...presetArgs, ...laterArgs);
    delete context[fnKey];
    return result;
  };
  return bound;
};

// 测试
function Person(name, age) { this.name = name; this.age = age; }
const BoundPerson = Person.myBind(null, 'Bob');
const p = new BoundPerson(25);
console.log(p.name, p.age);
输出 "Bob 25"，new 生效 ✅

Step 8: 修复原型链 — Object.create

困境：p instanceof Person 返回 false！因为 bound.prototype 没连接到 Person.prototype，原型链断了。
💭 怎么让 bound.prototype 继承 Person.prototype，又不直接引用它？
解法：Object.create(originalFn.prototype) 创建继承自原原型的新对象。不能用直接赋值，那样修改一方会污染另一方。
Function.prototype.myBind = function (thisArg, ...presetArgs) {
  const originalFn = this;
  const bound = function (...laterArgs) {
    const context = this instanceof bound ? this : thisArg;
    const fnKey = Symbol('fn');
    context[fnKey] = originalFn;
    const result = context[fnKey](...presetArgs, ...laterArgs);
    delete context[fnKey];
    return result;
  };
  if (originalFn.prototype) {
    bound.prototype = Object.create(originalFn.prototype);
  }
  return bound;
};

// 测试
function Person(name) { this.name = name; }
const BoundPerson = Person.myBind(null, 'Bob');
const p = new BoundPerson(25);
console.log(p instanceof Person);
输出 true，原型链修复 ✅

Step 9: name 属性 — Object.defineProperty

困境：原生 bind 返回的函数 name 带 "bound " 前缀，但我们的 bound 函数没有。函数的 name 默认不可写。
💭 函数的 name 属性默认 writable: false，怎么修改它？
解法：用 Object.defineProperty 强制设置 name，设为 configurable: true 保持和原生行为一致。
Function.prototype.myBind = function (thisArg, ...presetArgs) {
  const originalFn = this;
  const bound = function (...laterArgs) {
    const context = this instanceof bound ? this : thisArg;
    const fnKey = Symbol('fn');
    context[fnKey] = originalFn;
    const result = context[fnKey](...presetArgs, ...laterArgs);
    delete context[fnKey];
    return result;
  };
  if (originalFn.prototype) {
    bound.prototype = Object.create(originalFn.prototype);
  }
  Object.defineProperty(bound, 'name', {
    value: ('bound ' + (originalFn.name || '')).trim(),
    configurable: true,
  });
  return bound;
};

// 测试
function Person() {}
console.log(Person.myBind(null).name);
输出 "bound Person" ✅

面试追问速查

Q1：手写 new 操作符？
三步：创建空对象并链接原型 → 执行构造函数 → 如果返回值是对象就用它，否则用创建的对象。
Q2：箭头函数能用 bind 吗？
不能。箭头函数没有自己的 this，继承外层作用域，bind/call/apply 对它无效。
Q3：连续 bind 的优先级？
第一个 bind 生效，后续无法覆盖。内层的 thisArg 已硬绑定，再 bind 只是包一层壳。
Q4：严格模式的区别？
严格模式下 call(null) 传入的就是 null，不会回退到 globalThis。当前实现按非严格模式处理。
